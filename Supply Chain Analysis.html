#Helper Function: Distribution Plot

def plot_distribution(data, column, title, xlabel, ylabel):
    plt.figure(figsize=(8, 5))
    sns.histplot(data[column], bins=20, kde=True, color="blue")
    plt.title(title)
    plt.xlabel(xlabel)
    plt.ylabel(ylabel)
    plt.show()
